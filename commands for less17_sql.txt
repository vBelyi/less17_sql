SELECT COUNT(DISTINCT JOB_ID) FROM employees
2.
SELECT AVG(SALARY), COUNT(FIRST_NAME) AS QUANTITY_OF_EMPLOYEES FROM employees
WHERE DEPARTMENT_ID = 90
3.
SELECT JOB_ID, COUNT(*) AS QUANTITY_OF_EMPLOYEES
FROM employees
GROUP BY JOB_ID
HAVING QUANTITY_OF_EMPLOYEES > 1
4.
SELECT CONCAT(FIRST_NAME, ' ',LAST_NAME) AS EMPLOYEES_NAME, DEPARTMENT_ID, FIRST_NAME FROM employees
5.
SELECT CONCAT(FIRST_NAME, ' ',LAST_NAME) AS EMPLOYEES_NAME, JOBS.JOB_TITLE, DEPARTMENTS.DEPARTMENT_ID, FIRST_NAME, LOCATIONS.CITY
FROM EMPLOYEES
INNER JOIN JOBS ON JOBS.JOB_ID = EMPLOYEES.JOB_ID
INNER JOIN DEPARTMENTS ON DEPARTMENTS.DEPARTMENT_ID = EMPLOYEES.DEPARTMENT_ID
INNER JOIN LOCATIONS ON LOCATIONS.LOCATION_ID = DEPARTMENTS.LOCATION_ID
WHERE LOCATIONS.CITY = 'LONDON'
